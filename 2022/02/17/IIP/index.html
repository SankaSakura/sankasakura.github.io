% this file covers the use of imopen on the cells image
% analysis of dark_blobs is left as an exercise

im = imread(‘cell.jpg’);
imshow(im);
threshold = 210/255; % want to threshold at grey level 210, threshold needs to be between 0 and 1
BW = im2bw(im, threshold);
imshow(BW);
BW2 = imcomplement(BW); % need white foreground, black background
imshow(BW2);
se = strel(‘disk’,6); % make a circular structuring element
result = imopen(BW2, se);
imshow(result); % should see just the large cells

% reducing the element size gives most of the smaller cells too
se = strel(‘disk’,3);
result = imopen(BW2, se);
imshow(result);
% but also leaves some noise

% smaller cells are darker, should maybe change the threshold?
smallBW = im2bw(im,0.55);
smallBW2 = imcomplement(smallBW);
se = strel(‘disk’,2);
smallresult = imopen(smallBW2, se);
figure;
imshow(smallresult);
% gets all/most of the small cells, but leaves remnants of the large ones

% what if we combine the two results?
both = bitor(result, smallresult);
figure;
imshow(both);

im = imread(‘cameraman.tif’);
imshow(im);
h = images.roi.Circle(gca,’Center’,[100 100],’Radius’,50);
mask=createMask(h);
imshow(mask);
H = fspecial(‘gaussian’,20,5);
output = roifilt2(H, im, mask);
imshow(output);
im2=imread(‘eight.tif’);
imshow(im2);
T = graythresh(im2);
binary = imbinarize(im2, T);
imshow(binary);
mask2 = imcomplement(binary);
imshow(mask2);
smoothcoins = roifilt2(H, im2, mask2);
imshow(smoothcoins);
